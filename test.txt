#!/bin/bash

# تأكيد تشغيل السكربت بصلاحيات Root
if [ "$EUID" -ne 0 ]; then
  echo "[!] يرجى تشغيل السكربت باستخدام sudo"
  exit 1
fi

echo "[+] بدء تثبيت أشهر 20 أداة للشبكات والـ CTF على دبيان 13..."
apt update -y

# مصفوفة بالأدوات الأساسية التي يمكن تثبيتها مباشرة
basic_tools=(
    "nmap" "wireshark" "sqlmap" "john" "hashcat" 
    "hydra" "binwalk" "exiftool" "gobuster" "dirb" 
    "tmux" "curl" "wget" "tcpdump" "tshark" "aircrack-ng" 
    "crunch" "reaver" "pixiewps" "git" "python3-pip" 
    "python3-dev" "python3-full" "build-essential" "libpcap-dev"
)

echo "[+] 1. تثبيت الحزم الأساسية المتاحة في مستودعات دبيان..."
for tool in "${basic_tools[@]}"; do
    echo "[*] جاري محاولة تثبيت: $tool"
    DEBIAN_FRONTEND=noninteractive apt install -y $tool
done

echo "[+] 2. تثبيت الأدوات المتقدمة والواجهات الرسومية يدوياً لضمان نجاحها..."

# تثبيت Burp Suite Community (الواجهة الرسومية الشهيرة)
if ! command -v burpsuite &> /dev/null; then
    echo "[*] جاري تحميل وتثبيت Burp Suite..."
    wget -O /tmp/burp.sh "https://portswigger.net/burp/releases/startdownload?product=community&version=2026.5&type=Linux"
    chmod +x /tmp/burp.sh
    # التثبيت الصامت في الخلفية
    /tmp/burp.sh -q
    ln -sf /usr/local/bin/BurpSuiteCommunity /usr/local/bin/burpsuite 2>/dev/null || true
fi

# تثبيت Bettercap (أداة الشبكات المتقدمة)
if ! command -v bettercap &> /dev/null; then
    echo "[*] جاري تثبيت Bettercap وملحقاتها..."
    apt install -y bettercap
    bettercap -eval "caplets.update; q" 2>/dev/null || true
fi

# تثبيت Nikto (فحص ثغرات المواقع) من المصدر
if [ ! -d /opt/nikto ]; then
    echo "[*] جاري تثبيت Nikto يدوياً من GitHub..."
    git clone --depth 1 https://github.com/sullo/nikto.git /opt/nikto
    ln -sf /opt/nikto/program/nikto.pl /usr/local/bin/nikto
fi

# تثبيت Radare2 (الهندسة العكسية وحلول الـ CTF)
if ! command -v r2 &> /dev/null; then
    echo "[*] جاري بناء وتثبيت Radare2..."
    git clone --depth 1 https://github.com/radareorg/radare2.git /opt/radare2
    /opt/radare2/sys/install.sh
fi

# تثبيت مكتبة Pwntools الخاصة بالـ CTF للبايثون
echo "[*] تهيئة وتثبيت مكتبة Pwntools..."
python3 -m pip install pwntools --break-system-packages --quiet

echo "=================================================="
echo "[+] انتهت العملية! تفقد القائمة للتأكد من الأدوات."
echo "[+] وايرشارك و Burp Suite والـ 20 أداة جاهزة الآن للعمل."
echo "=================================================="
