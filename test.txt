#!/bin/bash

# تأكيد تشغيل السكربت بصلاحيات الجذور (Root)
if [ "$EUID" -ne 0 ]; then
  echo "[!] يرجى تشغيل السكربت باستخدام sudo (sudo bash install_tools.sh)"
  exit 1
fi

echo "[+] بدء تحديث وتجهيز نظام دبيان 13 للبيئة الأمنية..."
apt update -y

# 1. تثبيت الحزم والمكتبات الأساسية وأدوات وايرشارك وبيرب سويت وBettercap المتوفرة في دبيان
echo "[+] تثبيت الأدوات الأساسية ومكتبات البناء المتاحة..."
DEBIAN_FRONTEND=noninteractive apt install -y nmap dirb gobuster sqlmap john hashcat hydra binwalk exiftool curl wget tmux netcat-traditional xxd python3-pip python3-dev python3-full python3-venv git libssl-dev libffi-dev build-essential wireshark burpsuite aircrack-ng reaver pixiewps crunch bettercap libpcap-dev libnetfilter-queue-dev

# 2. حل مشكلة أداة Nikto وتثبيتها يدوياً لأنها محذوفة من مستودعات دبيان 13
if ! command -v nikto &> /dev/null; then
    echo "[+] أداة Nikto غير متوفرة في المستودعات، جاري تحميلها وتثبيتها من GitHub..."
    git clone --depth 1 https://github.com/sullo/nikto.git /opt/nikto
    ln -sf /opt/nikto/program/nikto.pl /usr/local/bin/nikto
fi

# 3. حل مشكلة أداة Radare2 وتثبيتها من المصدر
if ! command -v r2 &> /dev/null; then
    echo "[+] أداة Radare2 غير متوفرة في المستودعات، جاري تحميل النسخة الرسمية وتثبيتها..."
    git clone --depth 1 https://github.com/radareorg/radare2.git /opt/radare2
    /opt/radare2/sys/install.sh
fi

# 4. بناء هيكل المجلدات والـ Wordlists وتطابقها مع مسارات كالي الافتراضية
echo "[+] جاري إنشاء المجلدات وتحميل ملفات الـ Wordlists الحزمية..."
mkdir -p /usr/share/wordlists/dirb
mkdir -p /usr/share/wordlists/dirbuster

# تحميل وقذف ملف rockyou الحجم الكبير وفك ضغطه
if [ ! -f /usr/share/wordlists/rockyou.txt ]; then
    echo "[+] جاري تحميل ملف rockyou.txt..."
    wget -q -O /usr/share/wordlists/rockyou.txt.gz https://github.com/brannondorsey/naive-hashcat/releases/download/data/rockyou.txt.gz
    gzip -d /usr/share/wordlists/rockyou.txt.gz
fi

# تحميل قوائم فحص مسارات المواقع (Dirb)
wget -q -O /usr/share/wordlists/dirb/common.txt https://raw.githubusercontent.com/v0re/dirb/master/wordlists/common.txt
wget -q -O /usr/share/wordlists/dirb/big.txt https://raw.githubusercontent.com/v0re/dirb/master/wordlists/big.txt
wget -q -O /usr/share/wordlists/dirb/small.txt https://raw.githubusercontent.com/v0re/dirb/master/wordlists/small.txt

# تحميل قوائم Dirbuster
wget -q -O /usr/share/wordlists/dirbuster/directory-list-2.3-medium.txt https://raw.githubusercontent.com/daviddias/Words/master/Models/wordlists/dirbuster/directory-list-2.3-medium.txt

# تحميل حزمة SecLists الكاملة ووضعها في مكانها الصحيح مثل كالي
if [ ! -d /usr/share/wordlists/seclists ]; then
    echo "[+] جاري جلب وقذف حزمة SecLists بالكامل من GitHub..."
    git clone --depth 1 https://github.com/danielmiessler/SecLists.git /usr/share/wordlists/seclists
fi

# 5. تحديث كابليت Bettercap وتثبيت Pwntools
echo "[+] تحديث وتنشيط ميزات أداة Bettercap وتهيئة بيئة بايثون..."
bettercap -eval "caplets.update; q" 2>/dev/null || true

# تثبيت Pwntools بتخطي قيود بيئة دبيان المحمية
python3 -m pip install pwntools --break-system-packages --quiet

echo "=================================================="
echo "[+] اكتمل التثبيت بنجاح تام وبدون مشاكل!"
echo "[+] جميع الأدوات والـ Wordlists مرتبة في مسار /usr/share/wordlists/ وجاهزة للعمل."
echo "=================================================="
